Shape Drawing App

	CLASS:SHAPE

    ATTRIBUTES
    COLOR: TEXT
    POSITION: TEXT

    CONSTRUCTOR(C: TEXT, P: TEXT)
     SET COLOR TO C
     SET POSITION TO P

    END CONSTRUCTOR


    METHODS

        CALCULATE_AREA()
        DISPLAY "SHAPE AREA"
        END METHOD


        CALCULATE_PERIMETER()
        DISPLAY "SHAPE PERIMETER"
        END METHOD

END CLASS




CLASS CIRCLE INHERITS SHAPE

    ATTRIBUTES
    RADIUS: NUMBER
    CONSTRUCTOR(C: TEXT, P: TEXT, R: NUMBER)
    SUPER(C, P)
    SET RADIUS TO R
    END CONSTRUCTOR


    METHODS

        OVERRIDE CALCULATE_AREA()
        RETURN 3.14 * RADIUS * RADIUS
        END METHOD


        OVERRIDE CALCULATE_PERIMETER()
        RETURN 2 * 3.14 * RADIUS
        END METHOD

END CLASS




CLASS RECTANGLE INHERITS SHAPE

    ATTRIBUTES
        SIDE1: NUMBER
        SIDE2: NUMBER
        SIDE3: NUMBER


    CONSTRUCTOR(C: TEXT, P: TEXT, S1: NUMBER, S2: NUMBER, S3: NUMBER)
        SUPER(C, P)
        SET SIDE1 TO S1
        SET SIDE2 TO S2
        SET SIDE3 TO S3

    END CONSTRUCTOR


    METHODS

        OVERRIDE CALCULATE_AREA()
            SET S TO (SIDE1 + SIDE2 + SIDE3) / 2
            SET PART1 TO S - SIDE1
            SET PART2 TO S - SIDE2
            SET PART3 TO S - SIDE3
            SET PRODUCT TO S * PART1 * PART2 * PART3
            SET AREA TO SQRT(PRODUCT)
            RETURN AREA

        END METHOD

END CLASS




CLASS TRIANGLE INHERITS SHAPE

    ATTRIBUTES
        SIDE1: NUMBER
        SIDE2: NUMBER
        SIDE3: NUMBER


    CONSTRUCTOR(C: TEXT, P: TEXT, S1: NUMBER, S2: NUMBER, S3: NUMBER)
        SUPER(C, P)
        SET SIDE1 TO S1
        SET SIDE2 TO S2
        SET SIDE3 TO S3

    END CONSTRUCTOR


    METHODS

        OVERRIDE CALCULATE_AREA()
            SET S TO (SIDE1 + SIDE2 + SIDE3) / 2
            SET PART1 TO S - SIDE1
            SET PART2 TO S - SIDE2
            SET PART3 TO S - SIDE3
            SET PRODUCT TO S * PART1 * PART2 * PART3
            SET AREA TO SQRT(PRODUCT)
            RETURN AREA

        END METHOD


        OVERRIDE CALCULATE_PERIMETER()
            RETURN SIDE1 + SIDE2 + SIDE3
        END METHOD

END CLASS




CLASS CANVAS

    ATTRIBUTES
        SHAPES: LIST<SHAPE>


    CONSTRUCTOR()
        SET SHAPES TO LIST []

    END CONSTRUCTOR


    METHODS

        ADD_SHAPE(S: SHAPE)
            ADD S TO SHAPES
        END METHOD


        VIEW_SHAPES()

            SET I TO 0

            REPEAT WHILE I LESS THAN SIZE OF SHAPES

                SET SHAPE_NUMBER TO I + 1

                DISPLAY SHAPE_NUMBER + SHAPES[I].POSITION + SHAPES[I].COLOR

                DISPLAY "AREA = " + SHAPES[I].CALCULATE_AREA()

                DISPLAY "PERIMETER = " + SHAPES[I].CALCULATE_PERIMETER()

                SET I TO I + 1

            END REPEAT

        END METHOD

END CLASS




START

    BEGIN

        CREATE OBJECT CANVAS1 OF CANVAS()

        CREATE OBJECT CIRCLE1 OF CIRCLE("RED", "LEFT", 5)

        CREATE OBJECT RECT1 OF RECTANGLE("BLUE", "CENTER", 4, 6, 8)

        CREATE OBJECT TRI1 OF TRIANGLE("GREEN", "RIGHT", 3, 4, 5)

        CANVAS1.ADD_SHAPE(CIRCLE1)

        CANVAS1.ADD_SHAPE(RECT1)

        CANVAS1.ADD_SHAPE(TRI1)

        CANVAS1.VIEW_SHAPES()

    END

STOP
