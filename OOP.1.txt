START 
	BEGIN 
		Class Veterinary Clinic

		 ATTRIBUTES
		PRIVATE OWNERID :Number
		PRIVATE name: Text
		PRIVATE Pets: List <Text> []

		CONSTRUCTOR:(N,T:TEXT,P:[LIST]
		SET NAME TO N
		SET PETS TO P
		END STRUCTER

		METHOD
		ADD PET()
		REMOVE PET()
		GET PET()

		END METHOD
		CLASS PET
		PRIVATE PETID:NUMBER
		PRIVATE NAME:TEXT
		PRIVATE DATE_OF_BIRTH :TEXT
		PRIVATE MEDICALHISTORY :[LIST]
		
		CONSTRUCTOR:(N:TEXT,DOB:TEXT, MH:[LIST]
		SET NAME TO N
		SET DATE_OF_BIRTH TO DOB
		SET MEDICALHISTORY TO MH
		END CONSTRUCTER
		
		METHOD
		GET_NAME()
		 DISPLAY NAME
		END METHOD 
		GET_DATE_OF_BIRTH()
		DISPLAY DATE_OF_BIRTH
		END METHOD
		SET_MEDICALHISTORY(PROCEDURE:PET)
		ADD PROCEDURE TO LIST MEDICALHISTORY
		END METHOD

		CLASS EXAMINATION
		ATTRIBUTE
		PRIVATE EXAMINATIONID:NUMBER
		PRIVATE diagnosis: TEXT
		PRIVATE vital Signs:TEXT
		
		CONSTRUCTOR:(N,T:TEXT, D:TEXT, VS:TEXT]
		SET NAME TO N
		SET diagnosis TO D
		SET vital Signs TO VS
		END STRUCTER
		METHOD
		UPDATE RESULT()
		DISPLAY RESULT OF TRETMENTS 
		END METHOD

		CLASS TRETMENTS

		PRIVATE TRETMENTSID:NUMBER
		PRIVATE TRETMENTSNAME:TEXT
		PRIVATE description:TEXT
		PRIVATE COST:NUMBER
		
		CONSTRUCTOR:(N,T:TEXT, D:TEXT,C:NUMBER]
		SET TRETMENTSNAME TO N
		SET description TO D
		SET cost TO C
		END STRUCTER
		
		METHOD
		GET_COST()
	        DISPLAY COST 
		END METHOD
	END CLASS
