CLASS STORE
	ATTRIBUTE
		STORE_NAME:TEXT
		INSTRUMENTS:LIST"INSTRUMENT"

	CONSTRUCTER(SN:TEXT)
		SET STORE_NAME TO SN
		CREATE EMPTY LIST INSTRUMENTS
	END CONSTRUCTER

	METHOD
		ADD_INSTRUMENT(I:INSTRUMENT)
			ADD I TO INSTRUMENTS
		END METHOD

		REMOVE_INSTRUMENT(I:INSTRUMENT)
			REMOVE I FROM INSTRUMENTS
		END METHOD
END CLASS


START
BEGIN
	CREATE OBJECT STORE1 OF STORE("MUSIC WORLD")

	CREATE OBJECT GUITAR1 OF GUITAR("YAMAHA",250,6)
	CREATE OBJECT DRUM1 OF DRUM_KIT("PEARL",600,5)
	CREATE OBJECT KEYBOARD1 OF KEYBOARD("CASIO",350,88)

	STORE1.ADD_INSTRUMENT(GUITAR1)
	STORE1.ADD_INSTRUMENT(DRUM1)
	STORE1.ADD_INSTRUMENT(KEYBOARD1)

	GUITAR1.DESCRIBE_PLAY_STYLE()
	DRUM1.DESCRIBE_PLAY_STYLE()
	KEYBOARD1.DESCRIBE_PLAY_STYLE()
END
STOP
